在ForwardAdd 中设置 
Blend One One

// Attenuation 衰减计算
#ifdef USING_DIRECTIONAL_LIGHT
    fixed atten = 1.0;
#else
    #if defined (POINT)
        float3 lightCoord = mul(unity_WorldToLight, float4(i.worldPos, 1)).xyz;
        fixed atten = tex2D(_LightTexture0, dot(lightCoord, lightCoord).rr).UNITY_ATTEN_CHANNEL;
    #elif defined (SPOT)
        float4 lightCoord = mul(unity_WorldToLight, float4(i.worldPos, 1.0));
        fixed atten =   (lightCoord.z > 0) * 
                        tex2D(_LightTexture0, lightCoord.xy / lightCoord.w + 0.5).w * 
                        tex2D(_LightTextureB0, dot(lightCoord, lightCoord).rr).UNITY_ATTEN_CHANNEL;
    #else
        fixed atten = 1.0; 
    #endif

    // 或者自己算
    // float distance = length(_WorldSpaceLightPos0.xyz - i.worldPos);
    // fixed atten = 1.0 / distance;
#endif

return fixed4(ambient + (diffuse + specular) * atten, 1.0);

Shadow Caster 投射阴影

Pass
{
    Name "Shadow Caster"
    Tags
    {
        "LightMode" = "ShadowCaster"
    }

    CGPROGRAM
    #pragma multi_compile_shadowcaster
    #pragma vertex vert
    #pragma fragment frag
    #include "UnityCG.cginc"

    struct v2f
    {
        V2F_SHADOW_CASTER;
    };

    v2f vert(appdata_base v)
    {
        v2f o;
        TRANSFER_SHADOW_CASTER_NORMALOFFSET(o);
        return o;
    }

    float4 frag(v2f i) : SV_Target
    {
        SHADOW_CASTER_FRAGMENT(i);
    }
    ENDCG
}

// 单独做一个Material也是可以的，只是提示这样会消耗更多性能
// 建议在一个Shader中使用多Pass实现
// 更加建议使用Fallback实现

Shadow Receiver 阴影接收

// 在对应的地方加上代码即可

struct v2f
{
    SHADOW_COORDS(0) // 参数是下一个可用插值寄存器的索引值
};

v2f vert(a2v v)
{
    v2f ret;
    TRANSFER_SHADOW(ret);
    return ret;
}

fixed4 frag(v2f i) : SV_Target
{
    // Shadow Receiver
    fixed shadow = SHADOW_ATTENUATION(i);

    return fixed4(ambient + (diffuse + specular) * atten * shadow, 1.0);
}

同时包含光照 衰减 和 阴影 接收的宏
