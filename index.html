语义

struct appData
{
    // （模型空间）顶点位置 w = 1
    float4 vertex : POSITION;
    // （模型空间）顶点法线
    float3 normal : NORMAL;
    // （模型空间）顶点切线
    float3 tangent : TANGENT;
    // 模型的第一组纹理坐标
    float4 texcoord : TEXCOORD0;

struct v2f
{
    // Fragment 所需要的（裁剪空间）顶点位置
    float4 pos : SV_POSITION;

法线贴图的采样（示例）

fixed4 packedNormal = tex2D(_BumpMap,v_Data.uv.zw);
fixed3 tangentNormal;

// BumpMap is not marked as "Normal Map"
// 手动解析xy。不要使用，手动解析没有区分贴图类型选择正确的算法
// tangentNormal.xy = (packedNormal.xy * 2 - 1);

// BumpMap is marked as "Normal Map"
tangentNormal.xy = UnpackNormal(packedNormal);      // build-in function "UnpackNormal()"
tangentNormal.xy *= _BumpScale;
tangentNormal.z = sqrt(1.0 - saturate(dot(tangentNormal.xy, tangentNormal.xy)));    // v·v = |v|^2

// UnpackNormal(BumpMap的采样后解析) 
=> UnityCG.cginc

任意 A空间 <== B空间 的转换

需要知道 A空间下，B坐标系的xyz轴向量 和 B原点坐标
并以*列*矩阵排序，举例如下：
// 注释： B_X_y 为 在A空间中，B坐标系的X轴向量的y分量，O为原点
Matrix_BA = ( B_X_x, B_Y_x, B_Z_x, B_O_x )
            ( B_X_y, B_Y_y, B_Z_y, B_O_y )
            ( B_X_z, B_Y_z, B_Z_z, B_O_z )
            ( 0,     0,     0,     1     )

仅可以得到 Matrix_BA B=>A 的 变换矩阵

// 特别说明
1 : 当使用变换矩阵对*方向向量*进行变换时，是不需要关心位移的
    所以使用 (float3x3)Mtrix_BA 即可
2 : 当使用的时 (float3x3)Matrix_BA, 并且用于构建该变换矩阵的 xyz轴向量是单位向量时
    (float3x3)Matrix_BA 为正交矩阵，由正交矩阵的特性可得
    (float3x3)Matrix_BA.Tranpose == (float3x3)Matrix_BA.Inverse == (float3x3)Matrix_AB
